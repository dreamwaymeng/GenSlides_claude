                \fill[black!30] (0,0) rectangle (#1\textwidth, 1pt);
                \fill[themecolorfg] (0,0) rectangle (#1\textwidth/2, 1pt);
            \end{tikzpicture}
        \end{tabular}
    \end{frame}
}
